# 饮料瓶收集合成游戏 V0 实现计划

> **面向 AI 代理的工作者：** 使用 subagent-driven-development 逐任务实现此计划。步骤使用复选框（`- [ ]`）语法来跟踪进度。

**目标：** 构建一个可在浏览器中运行的像素风饮料瓶收集合成游戏原型，包含4台贩卖机、抽卡、背包、三合一合成、图鉴、金币经济和限定瓶系统。

```javascript
  currentPage: 'machine',

// 页面注册表
const pages = {};

export function registerPage(name, renderFn) {
  pages[name] = renderFn;
}

function switchPage(name) {
  state.currentPage = name;
  document.querySelectorAll('#tab-bar .tab').forEach(t => {
    t.classList.toggle('active', t.dataset.page === name);
  });
  render();
}

export function render() {
  const container = document.getElementById('page-container');
  const fn = pages[state.currentPage];
  if (fn) {
    container.innerHTML = '';
    fn(container);
  }
}

// Tab 点击事件
document.querySelectorAll('#tab-bar .tab').forEach(tab => {
  tab.addEventListener('click', () => switchPage(tab.dataset.page));
});

// 初始渲染
```

```javascript
// js/utils/random.js

export function weightedRandom(items, weightFn) {
  const totalWeight = items.reduce((sum, item) => sum + weightFn(item), 0);
  let r = Math.random() * totalWeight;
  for (const item of items) {
    r -= weightFn(item);
    if (r <= 0) return item;
  }
  return items[items.length - 1];
}

export function randomInt(min, max) {
  return Math.floor(Math.random() * (max - min + 1)) + min;
}

export function chance(probability) {
  return Math.random() < probability;
}

export function shuffle(arr) {
  const a = [...arr];
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}
```

```javascript
// js/services/dataAdapter.js

const STORAGE_KEY = 'bottle_game_save';

// 默认玩家初始数据
const DEFAULT_PLAYER = {
  coins: 200,
  bottles: {},        // { bottleId: count }
  machines: ['vm_soda'], // 初始解锁
  ownedNewCount: 0,   // 累计获得"新种类"数量（用于阶段盲抽）
  blindBoxUsed: 0,    // 已使用的阶段盲抽次数
  pityCounts: { 3: 0, 4: 0, 5: 0 }, // 保底计数器（按稀有度）
};

/**
 * DataAdapter — 抽象数据读写接口
 * V0: localStorage
 * V1: 替换为云数据库适配器
 */
export const dataAdapter = {
  load() {
    try {
      const raw = localStorage.getItem(STORAGE_KEY);
      if (raw) {
        const data = JSON.parse(raw);
        // 合并默认值，防止新增字段缺失
        return { ...DEFAULT_PLAYER, ...data };
      }
    } catch (e) {
      console.warn('读取存档失败，使用默认数据', e);
    }
    return { ...DEFAULT_PLAYER };
  },

  save(playerData) {
    try {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(playerData));
    } catch (e) {
      console.error('保存存档失败', e);
    }
  },

  reset() {
    localStorage.removeItem(STORAGE_KEY);
  },
};

// 获取当前玩家数据（内存缓存，页面间共享）
let _player = null;

export function getPlayer() {
  if (!_player) {
    _player = dataAdapter.load();
  }
  return _player;
}

export function savePlayer() {
  if (_player) {
    dataAdapter.save(_player);
  }
}
```

```javascript
// js/services/economyService.js
import { getPlayer, savePlayer } from './dataAdapter.js';
import { SELL_PRICES } from '../utils/constants.js';

export const economyService = {

  /** 获取当前金币 */
  getCoins() {
    return getPlayer().coins;
  },

  /** 增加金币（签到/奖励/卖出等） */
  addCoins(amount) {
    const p = getPlayer();
    p.coins += amount;
    savePlayer();
    return p.coins;
  },

  /** 消费金币，返回是否成功 */
  spendCoins(amount) {
    const p = getPlayer();
    if (p.coins < amount) return false;
    p.coins -= amount;
    savePlayer();
    return true;
  },

    if (p.bottles[bottleId] <= 0) delete p.bottles[bottleId];
```

```javascript
// js/services/machineService.js
import { getPlayer, savePlayer } from './dataAdapter.js';
import { MACHINES, getMachineById } from '../data/machines.js';
import { BOTTLES } from '../data/bottles.js';
import { bottleService } from './bottleService.js';
import { economyService } from './economyService.js';
import { weightedRandom, chance } from '../utils/random.js';
import { RARITY, AGAIN_ONE_PROB, PITY_COUNTS, PULL_COST } from '../utils/constants.js';

/** 基础概率权重 */
const BASE_WEIGHTS = {
  [RARITY.COMMON]: 545,
  [RARITY.RARE]: 300,
  [RARITY.PRECIOUS]: 120,
  [RARITY.EPIC]: 30,
  [RARITY.LEGENDARY]: 5,
};

export const machineService = {

  /** 获取已解锁贩卖机列表 */
  getUnlockedMachines() {
    const p = getPlayer();
    return MACHINES.filter(m => p.machines.includes(m.id));
  },

  /** 解锁贩卖机 */
  unlockMachine(machineId) {
    const p = getPlayer();
    const machine = getMachineById(machineId);
    if (!machine) return { success: false, reason: '贩卖机不存在' };
    if (p.machines.includes(machineId)) return { success: false, reason: '已解锁' };

    if (machine.unlockType === 'initial') {
      return { success: false, reason: '初始贩卖机无需解锁' };
    }

    if (machine.unlockType === 'coin') {
      if (p.coins < machine.unlockCost) {
        return { success: false, reason: `金币不足（需要 ${machine.unlockCost}）` };
      }
      p.coins -= machine.unlockCost;
      p.machines.push(machineId);
      savePlayer();
      return { success: true };
    }

    return { success: false, reason: '未知解锁方式' };
  },

  /** 检查贩卖机是否已解锁 */
  isUnlocked(machineId) {
    return getPlayer().machines.includes(machineId);
  },

  /**
   * 抽卡 — 核心方法
   * @param {string} machineId 贩卖机ID
   * @param {number} times 抽卡次数 (1 或 10)
   * @returns {{ success, bottles: [], cost, newBottles: [], againOne: boolean }}
   */
  pull(machineId, times = 1) {
    const machine = getMachineById(machineId);
    if (!machine) return { success: false, reason: '贩卖机不存在' };
    if (!this.isUnlocked(machineId)) return { success: false, reason: '贩卖机未解锁' };

    const cost = times === 1 ? machine.pullCost : machine.pullTenCost;
    if (!economyService.spendCoins(cost)) {
      return { success: false, reason: `金币不足（需要 ${cost}）` };
    }

    const pool = BOTTLES.filter(machine.poolFilter);
    if (pool.length === 0) return { success: false, reason: '奖池为空' };

    const results = [];
    const newBottles = [];
    let gotAgainOne = false;

    for (let i = 0; i < times; i++) {
      const bottle = this._rollBottle(pool, machineId);
      if (!bottle) continue;

      const addResult = bottleService.addBottle(bottle.id);
      results.push(bottle);
      if (addResult.isNew) newBottles.push(bottle);

      // 再来一瓶判定
      if (chance(AGAIN_ONE_PROB)) {
        gotAgainOne = true;
        const extra = this._rollBottle(pool, machineId);
        if (extra) {
          const extraResult = bottleService.addBottle(extra.id);
          results.push(extra);
          if (extraResult.isNew) newBottles.push(extra);
        }
      }
    }

    savePlayer();
    return { success: true, bottles: results, cost, newBottles, againOne: gotAgainOne };
  },

  /** 内部：按概率+保底抽取一个瓶子 */
  _rollBottle(pool, machineId) {
    const p = getPlayer();

    // 按稀有度分组
    const byRarity = {};
    for (const b of pool) {
      if (!byRarity[b.rarity]) byRarity[b.rarity] = [];
      byRarity[b.rarity].push(b);
    }

    // 确定目标稀有度
    let targetRarity;
    // 检查保底
    if (p.pityCounts[5] >= PITY_COUNTS[5] - 1 && byRarity[5]) {
      targetRarity = 5;
      p.pityCounts[5] = 0;
    } else if (p.pityCounts[4] >= PITY_COUNTS[4] - 1 && byRarity[4]) {
      targetRarity = 4;
      p.pityCounts[4] = 0;
    } else if (p.pityCounts[3] >= PITY_COUNTS[3] - 1 && byRarity[3]) {
      targetRarity = 3;
      p.pityCounts[3] = 0;
    } else {
      // 常规概率
      const rarities = Object.keys(byRarity).map(Number).filter(r => byRarity[r].length > 0);
      const entries = rarities.map(r => ({ rarity: r, weight: BASE_WEIGHTS[r] || 0 }));
      if (entries.length === 0) return null;
      targetRarity = weightedRandom(entries, e => e.weight).rarity;
    }

    // 更新保底计数器
    for (const r of [3, 4, 5]) {
      if (r > targetRarity) {
        p.pityCounts[r] = Math.min((p.pityCounts[r] || 0) + 1, PITY_COUNTS[r] || 999);
      } else if (r <= targetRarity) {
        p.pityCounts[r] = 0;
      }
    }

    // 从目标稀有度中随机选一个
    const candidates = byRarity[targetRarity];
    return candidates[Math.floor(Math.random() * candidates.length)];
  },
};
```

```javascript
import { BOTTLES } from '../data/bottles.js';
import { bottleService } from './bottleService.js';
import { economyService } from './economyService.js';
import { SYNTH_COSTS } from '../utils/constants.js';

export const synthService = {

  /**
   * 三合一合成
   * @param {string} bottleId — 消耗3个该ID的瓶子
   * @returns {{ success, newBottle?, cost }}
   */
  synthUpgrade(bottleId) {
    const source = BOTTLES.find(b => b.id === bottleId);
    if (!source) return { success: false, reason: '瓶子不存在' };
    if (source.limited) return { success: false, reason: '限定瓶不可用于合成' };
    if (bottleService.getCount(bottleId) < 3) return { success: false, reason: '需要3个相同瓶子' };
    if (source.rarity >= 5) return { success: false, reason: '传说已是最高稀有度' };

    const targetRarity = source.rarity + 1;
    const cost = SYNTH_COSTS[source.rarity] || 200;
    if (!economyService.spendCoins(cost)) {
      return { success: false, reason: `金币不足（需要 ${cost}）` };
    }

    // 移除3个源瓶子
    bottleService.removeBottle(bottleId, 3);

    // 随机获得一个目标稀有度的普通瓶
    const candidates = BOTTLES.filter(b => b.rarity === targetRarity && !b.limited);
    if (candidates.length === 0) return { success: false, reason: '没有可合成的目标瓶子' };

    const newBottle = candidates[Math.floor(Math.random() * candidates.length)];
    bottleService.addBottle(newBottle.id);

    return { success: true, newBottle, cost };
  },

  /**
   * 阶段盲抽
   * @returns {{ success, newBottle? }}
   */
  blindBoxPull() {
    if (!bottleService.checkBlindBox()) return { success: false, reason: '暂无盲抽资格' };

    bottleService.useBlindBox();

    // 盲抽池：⭐⭐⭐+ 瓶子（含极小概率出限定）
    const pool = BOTTLES.filter(b => b.rarity >= 3);
    if (pool.length === 0) return { success: false, reason: '奖池为空' };

    const bottle = pool[Math.floor(Math.random() * pool.length)];
    bottleService.addBottle(bottle.id);

    return { success: true, newBottle: bottle };
  },
};
```

```javascript
/**
 * 在 Canvas 上绘制像素风饮料瓶
 * @param {CanvasRenderingContext2D} ctx
 * @param {number} x - 画布中的x坐标
 * @param {number} y - 画布中的y坐标

export function drawBottleSprite(ctx, x, y, bottle, scale = 1) {
  const size = S * scale;
  ctx.save();
  ctx.imageSmoothingEnabled = false;

  // 瓶子外框（稀有度颜色）
  const borderColor = bottle.limited ? '#FFD700' : (RARITY_COLORS[bottle.rarity] || '#888');
  ctx.fillStyle = '#1a1a2e';
  ctx.fillRect(x - 1, y - 1, size + 2, size + 2);

  // 瓶子底色
  ctx.fillStyle = bottle.spriteColor;
  const shape = bottle.shape || 'can';
  const padding = 3 * scale;
  const px = x + padding, py = y + padding;
  const pw = size - padding * 2, ph = size - padding * 2;

  switch (shape) {
    case 'can': drawCan(ctx, px, py, pw, ph); break;
    case 'glass': drawGlass(ctx, px, py, pw, ph); break;
    case 'bottle': drawBottleShape(ctx, px, py, pw, ph); break;
    case 'cup': drawCup(ctx, px, py, pw, ph); break;
    case 'tetra': drawTetra(ctx, px, py, pw, ph); break;
    default: drawCan(ctx, px, py, pw, ph);
  }

  // 稀有度边框
  ctx.strokeStyle = borderColor;
  ctx.lineWidth = 2 * scale;
  ctx.strokeRect(x, y, size, size);

  // 传说光效
  if (bottle.rarity === RARITY.LEGENDARY) {
    ctx.strokeStyle = '#FFD700';
    ctx.lineWidth = 2 * scale;
    ctx.shadowColor = '#FFD700';
    ctx.shadowBlur = 6 * scale;
    ctx.strokeRect(x + 1, y + 1, size - 2, size - 2);
    ctx.shadowBlur = 0;

    ctx.lineWidth = 1 * scale;
    ctx.setLineDash([3 * scale, 3 * scale]);

    ctx.setLineDash([]);

  ctx.restore();
}
```
